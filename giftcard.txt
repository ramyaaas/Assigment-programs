using System;

using System.Text;

namespace Giftcard1
{
    public class gift
    {
        private long cardNo;
        private double balance;
        private Boolean active;
        

        public long Cardno
        {
            get { return cardNo; }
            set { cardNo = value; }
        }

        public Boolean Active
        {
            get { return active; }
            set { active = value; }
        }

        public double Balance
        {
            get { return balance; }
        }
        public gift(long no, Boolean active)
        {
            this.cardNo = no;
            //this.balance = balance;
            this.active = active;
        }
        public gift()
        {
            this.balance = 0;
            this.active = true;
        }

        public void SwipeCard(double amt)
        {
            if(active==true && balance>amt)
            {
                 balance=balance-amt;
            }
            

            else
            {
                Console.WriteLine("the account is not contains manimum balance");
            }
            Console.WriteLine("the available balance is {0}", balance);
        }
        public void RechargeCard(double amt)
        {
            if (active == true)
            {
                balance = balance + amt;
            }
            else
            {
                Console.WriteLine("the account is not active");
            }
        }
        
    }
    

        
}
